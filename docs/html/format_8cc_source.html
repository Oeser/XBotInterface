143  // Handle the result of XSI-compliant version of strerror_r.
144  int handle(int result)
145  {
146  // glibc versions before 2.13 return result in errno.
147  return result == -1 ? errno : result;
148  }
149 
150  // Handle the result of GNU-specific version of strerror_r.
151  int handle(char *message)
152  {
153  // If the buffer is full then the message is probably truncated.
154  if (message == buffer_ && strlen(buffer_) == buffer_size_ - 1)
155  return ERANGE;
156  buffer_ = message;
157  return 0;
158  }
159 
160  // Handle the case when strerror_r is not available.
161  int handle(internal::Null<>)
162  {
163  return fallback(strerror_s(buffer_, buffer_size_, error_code_));
164  }
165 
166  // Fallback to strerror_s when strerror_r is not available.
167  int fallback(int result)
168  {
169  // If the buffer is full then the message is probably truncated.
170  return result == 0 && strlen(buffer_) == buffer_size_ - 1 ?
171  ERANGE : result;
172  }
173 
174  // Fallback to strerror if strerror_r and strerror_s are not available.
175  int fallback(internal::Null<>)
176  {
177  errno = 0;
178  buffer_ = strerror(error_code_);
179  return errno;
180  }
181 
182  public:
183  StrError(int err_code, char *&buf, std::size_t buf_size)
184  : error_code_(err_code), buffer_(buf), buffer_size_(buf_size)
185  {}

187  int run()
188  {
189  // Suppress a warning about unused strerror_r.
190  strerror_r(0, FMT_NULL, "");
191  return handle(strerror_r(error_code_, buffer_, buffer_size_));
192  }
193  };

197  void format_error_code(Writer &out, int error_code,
198  StringRef message) FMT_NOEXCEPT
199  {
200  // Report error code making sure that the output fits into
201  // INLINE_BUFFER_SIZE to avoid dynamic memory allocation and potential
202  // bad_alloc.
203  out.clear();
204  static const char SEP[] = ": ";
205  static const char ERROR_STR[] = "error ";
206  // Subtract 2 to account for terminating null characters in SEP and ERROR_STR.
207  std::size_t error_code_size = sizeof(SEP) + sizeof(ERROR_STR) - 2;
208  typedef internal::IntTraits<int>::MainType MainType;
209  MainType abs_value = static_cast<MainType>(error_code);
210  if (internal::is_negative(error_code)) {
211  abs_value = 0 - abs_value;
212  ++error_code_size;
213  }
214  error_code_size += internal::count_digits(abs_value);
215  if (message.size() <= internal::INLINE_BUFFER_SIZE - error_code_size)
216  out << message << SEP;
217  out << ERROR_STR << error_code;
218  assert(out.size() <= internal::INLINE_BUFFER_SIZE);
219  }

386  FMT_FUNC void internal::format_windows_error(
387  Writer &out, int error_code, StringRef message) FMT_NOEXCEPT
388  {
389  FMT_TRY{
390  MemoryBuffer<wchar_t, INLINE_BUFFER_SIZE> buffer;
391  buffer.resize(INLINE_BUFFER_SIZE);
392  for (;;) {
393  wchar_t *system_message = &buffer[0];
394  int result = FormatMessageW(
395  FORMAT_MESSAGE_FROM_SYSTEM | FORMAT_MESSAGE_IGNORE_INSERTS,
396  FMT_NULL, error_code, MAKELANGID(LANG_NEUTRAL, SUBLANG_DEFAULT),
397  system_message, static_cast<uint32_t>(buffer.size()), FMT_NULL);

399  UTF16ToUTF8 utf8_message;

404  break;
405  }
