205 bool XBot::ModelInterface::fillModelOrderedChain()
206 {
207 
208  bool success = true;
209 
210  std::vector<std::string> model_ordered_joint_name;
211  this->getModelOrderedJoints(model_ordered_joint_name);
212 
213  // if model is floating base add virtual chain with six virtual joints,
214  // which are the first six provided by getModelOrderedJoints
215 
216  int joint_idx = 0;
217  _joint_id_to_model_id.clear();
218  if(isFloatingBase()) {
219 
220  std::string virtual_chain_name("virtual_chain");
221 
222  _chain_map[virtual_chain_name] = std::make_shared<XBot::KinematicChain>(virtual_chain_name);
223 
224  for(int i=0; i<6; i++){
225 
226  boost::shared_ptr<urdf::Joint> virtual_joint(new urdf::Joint);
227  virtual_joint->type = urdf::Joint::UNKNOWN;
228  virtual_joint->limits.reset(new urdf::JointLimits);
229  virtual_joint->limits->effort = 1e9;
230  virtual_joint->limits->lower = -1e9;
231  virtual_joint->limits->upper = +1e9;
232  virtual_joint->limits->velocity = 1e9;
233 
234 
235  XBot::Joint::Ptr jptr = std::make_shared<Joint>(model_ordered_joint_name[i],
236  -(i+1),
237  virtual_joint,
238  virtual_chain_name);
239 
240  _chain_map.at(virtual_chain_name)->pushBackJoint(jptr);
241 
242  _joint_id_to_model_id[jptr->getJointId()] = joint_idx;
243 
244  joint_idx++;
245  }
246 
247 // _model_ordered_chain_name.push_back(virtual_chain_name);
248  }
249 
250 
251  _ordered_chain_names.clear();
252  joint_idx = 0;
253  while( joint_idx < model_ordered_joint_name.size() ){
254 
255  // compute the chain which the joint being processed belongs to
256  std::string joint_name = model_ordered_joint_name[joint_idx];
257  XBot::Joint::ConstPtr joint_ptr = getJointByName(joint_name);
258  if(!joint_ptr) {
259  return false;
260  }
261 
262  std::string chain_name = joint_ptr->getChainName();
263 
264  _ordered_chain_names.push_back(chain_name);
265 
266  // check that the joint that follow are equal to the chain ones,
267  // ordered from base link to tip link
268  const XBot::KinematicChain& chain = *_chain_map.at(chain_name);
269 
270  int chain_joint_num = chain.getJointNum();
271 
272  for( int i = 0; i < chain_joint_num; i++ ){
273 
274  if( chain.getJointName(i) == model_ordered_joint_name[joint_idx] ){
275 
276  _joint_id_to_model_id[chain.getJointId(i)] = joint_idx;
277  joint_idx++;
278 
279  }
280  else{
281  return false;
282  }
283 
284  }
285 
286 
287  }
288 
289  //_model_ordered_chain_name;
290 
291  std::cout << "Model ordered chains: " << std::endl;
292  for(const auto& s : _ordered_chain_names) std::cout << s << std::endl;
293 
294  return success;
295 }

418 bool XBot::ModelInterface::getRelativeJacobian(const std::string& target_link_name,
419  const std::string& base_link_name,
420  KDL::Jacobian& J) const
421 {
422  /* Define some alias for easier reading */
423  KDL::Jacobian& base_jacobian = _tmp_kdl_jacobian;
424  KDL::Jacobian& target_jacobian = _tmp_kdl_jacobian_1;
425  KDL::Rotation& world_R_base = _tmp_kdl_rotation;
426  KDL::Vector& world_deltap_21 = _tmp_kdl_vector;
427 
428  /* Compute base e target link jacobians */
429  bool success = getJacobian(base_link_name, base_jacobian);
430  success = getJacobian(target_link_name, target_jacobian) && success;
431 
432  /* Compute orientation of base link w.r.t world */
433  success = getOrientation(base_link_name, world_R_base) && success;
434 
435  /* Compute the vector origin_target-origin_base w.r.t. world */
436  KDL::Vector world_origin_base, world_origin_target;
437  getPointPosition(base_link_name, KDL::Vector::Zero(), world_origin_base);
438  getPointPosition(target_link_name, KDL::Vector::Zero(), world_origin_target);
439  world_deltap_21 = world_origin_target - world_origin_base;
440 
441  /* Change ref point of base jacobian from origin_base to origin_target */
442  base_jacobian.changeRefPoint(world_deltap_21);
443 
444  /* Compute relative jacobian in world frame */
445  J.data = target_jacobian.data - base_jacobian.data;
446 
447  /* Rotate to base frame */
448  J.changeBase(world_R_base.Inverse());
449 
450 
451  return success;
452 }

700 bool XBot::ModelInterface::computeConstrainedInverseDynamics(const Eigen::MatrixXd& J,
701  const Eigen::VectorXd& weights,
702  Eigen::VectorXd& tau) const
703 {
704  int num_constraints = J.rows();
705  Eigen::VectorXd _h, _b, _w;
706  Eigen::MatrixXd _Q, _A;
707  computeInverseDynamics(_h);
708 
709  Eigen::HouseholderQR<Eigen::MatrixXd> _qr(J.transpose());
710  _Q = _qr.householderQ();
711 
712  _b = (_Q.transpose() * _h).tail(getJointNum() - num_constraints);
713  _A = _Q.bottomRows(getJointNum()).transpose().bottomRows(getJointNum() - num_constraints);
714 
715  _w = weights.array().inverse().sqrt();
716 
717  _A.noalias() = _A*_w.asDiagonal();
718 
719  tau = _w.asDiagonal() * _A.jacobiSvd(Eigen::ComputeThinU|Eigen::ComputeThinV).solve(_b);
720 
721 }
