234 inline void spdlog::details::async_log_helper::push_msg(details::async_log_helper::async_msg&& new_msg)
235 {
236  if (!_q.enqueue(std::move(new_msg)) && _overflow_policy != async_overflow_policy::discard_log_msg)
237  {
238  auto last_op_time = details::os::now();
239  auto now = last_op_time;
240  do
241  {
242  now = details::os::now();
243  sleep_or_yield(now, last_op_time);
244  }
245  while (!_q.enqueue(std::move(new_msg)));
246  }
247 }

257 inline void spdlog::details::async_log_helper::worker_loop()
258 {
259  try
260  {
261  if (_worker_warmup_cb) _worker_warmup_cb();
262  auto last_pop = details::os::now();
263  auto last_flush = last_pop;
264  while(process_next_msg(last_pop, last_flush));
265  if (_worker_teardown_cb) _worker_teardown_cb();
266  }
267  catch (const std::exception &ex)
268  {
269  _err_handler(ex.what());
270  }
271  catch (...)
272  {
273  _err_handler("Unknown exception");
274  }
275 }

277 // process next message in the queue
278 // return true if this thread should still be active (while no terminate msg was received)
279 inline bool spdlog::details::async_log_helper::process_next_msg(log_clock::time_point& last_pop, log_clock::time_point& last_flush)
280 {
281  async_msg incoming_async_msg;
282 
283  if (_q.dequeue(incoming_async_msg))
284  {
285  last_pop = details::os::now();
286  switch (incoming_async_msg.msg_type)
287  {
288  case async_msg_type::flush:
289  _flush_requested = true;
290  break;
291 
292  case async_msg_type::terminate:
293  _flush_requested = true;
294  _terminate_requested = true;
295  break;
296 
297  default:
298  log_msg incoming_log_msg;
299  incoming_async_msg.fill_log_msg(incoming_log_msg);
300  _formatter->format(incoming_log_msg);
301  for (auto &s : _sinks)
302  {
303  if(s->should_log( incoming_log_msg.level))
304  {
305  s->log(incoming_log_msg);
306  }
307  }
308  }
309  return true;
310  }
311 
312  // Handle empty queue..
313  // This is the only place where the queue can terminate or flush to avoid losing messages already in the queue
314  else
315  {
316  auto now = details::os::now();
317  handle_flush_interval(now, last_flush);
318  sleep_or_yield(now, last_pop);
319  return !_terminate_requested;
320  }
321 }

323 // flush all sinks if _flush_interval_ms has expired
324 inline void spdlog::details::async_log_helper::handle_flush_interval(log_clock::time_point& now, log_clock::time_point& last_flush)
325 {
326  auto should_flush = _flush_requested || (_flush_interval_ms != std::chrono::milliseconds::zero() && now - last_flush >= _flush_interval_ms);
327  if (should_flush)
328  {
329  for (auto &s : _sinks)
330  s->flush();
331  now = last_flush = details::os::now();
332  _flush_requested = false;
333  }
334 }

342 // spin, yield or sleep. use the time passed since last message as a hint
343 inline void spdlog::details::async_log_helper::sleep_or_yield(const spdlog::log_clock::time_point& now, const spdlog::log_clock::time_point& last_op_time)
344 {
345  using namespace std::this_thread;
346  using std::chrono::milliseconds;
347  using std::chrono::microseconds;
348 
349  auto time_since_op = now - last_op_time;
350 
351  // spin upto 50 micros
352  if (time_since_op <= microseconds(50))
353  return;
354 
355  // yield upto 150 micros
356  if (time_since_op <= microseconds(100))
357  return std::this_thread::yield();
358 
359  // sleep for 20 ms upto 200 ms
360  if (time_since_op <= milliseconds(200))
361  return sleep_for(milliseconds(20));
362 
363  // sleep for 200 ms
364  return sleep_for(milliseconds(200));
365 }
