397 // Standard permits specialization of std::numeric_limits. This specialization
398 // is used to resolve ambiguity between isinf and std::isinf in glibc:
399 // https://gcc.gnu.org/bugzilla/show_bug.cgi?id=48891
400 // and the same for isnan and signbit.
401 template <>
402 class numeric_limits<fmt::internal::DummyInt>:
403  public std::numeric_limits<int>
404 {
405 public:
406  // Portable version of isinf.
407  template <typename T>
408  static bool isinfinity(T x)
409  {
410  using namespace fmt::internal;
411  // The resolution "priority" is:
412  // isinf macro > std::isinf > ::isinf > fmt::internal::isinf
413  if (const_check(sizeof(isinf(x)) == sizeof(bool) ||
414  sizeof(isinf(x)) == sizeof(int)))
415  {
416  return isinf(x) != 0;
417  }
418  return !_finite(static_cast<double>(x));
419  }
420 
421  // Portable version of isnan.
422  template <typename T>
423  static bool isnotanumber(T x)
424  {
425  using namespace fmt::internal;
426  if (const_check(sizeof(isnan(x)) == sizeof(bool) ||
427  sizeof(isnan(x)) == sizeof(int)))
428  {
429  return isnan(x) != 0;
430  }
431  return _isnan(static_cast<double>(x)) != 0;
432  }
433 
434  // Portable version of signbit.
435  static bool isnegative(double x)
436  {
437  using namespace fmt::internal;
438  if (const_check(sizeof(signbit(x)) == sizeof(bool) ||
439  sizeof(signbit(x)) == sizeof(int)))
440  {
441  return signbit(x) != 0;
442  }
443  if (x < 0) return true;
444  if (!isnotanumber(x)) return false;
445  int dec = 0, sign = 0;
446  char buffer[2]; // The buffer size must be >= 2 or _ecvt_s will fail.
447  _ecvt_s(buffer, sizeof(buffer), x, 0, &dec, &sign);
448  return sign != 0;
449  }

822 // SIZE elements stored in the object itself.
823 template <typename T, std::size_t SIZE, typename Allocator = std::allocator<T> >
824 class MemoryBuffer: private Allocator, public Buffer<T>
825 {
826 private:
827  T data_[SIZE];
828 
829  // Deallocate memory allocated by the buffer.
830  void deallocate()
831  {
832  if (this->ptr_ != data_) Allocator::deallocate(this->ptr_, this->capacity_);
833  }
834 
835 protected:
836  void grow(std::size_t size) FMT_OVERRIDE;
837 
838 public:
839  explicit MemoryBuffer(const Allocator &alloc = Allocator())
840  : Allocator(alloc), Buffer<T>(data_, SIZE)
841  {}
842  ~MemoryBuffer()
843  {
844  deallocate();
845  }
846 
847 #if FMT_USE_RVALUE_REFERENCES
848 private:
849  // Move data from other to this buffer.
850  void move(MemoryBuffer &other)
851  {
852  Allocator &this_alloc = *this, &other_alloc = other;
853  this_alloc = std::move(other_alloc);
854  this->size_ = other.size_;
855  this->capacity_ = other.capacity_;
856  if (other.ptr_ == other.data_)
857  {
858  this->ptr_ = data_;
859  std::uninitialized_copy(other.data_, other.data_ + this->size_,
860  make_ptr(data_, this->capacity_));
861  }
862  else
863  {
864  this->ptr_ = other.ptr_;
865  // Set pointer to the inline array so that delete is not called
866  // when deallocating.
867  other.ptr_ = other.data_;
868  }
869  }
870 
871 public:
872  MemoryBuffer(MemoryBuffer &&other)
873  {
874  move(other);
875  }
876

878  {
879  assert(this != &other);
880  deallocate();
881  move(other);
882  return *this;
883  }
884 #endif

890  }

893 template <typename T, std::size_t SIZE, typename Allocator>
894 void MemoryBuffer<T, SIZE, Allocator>::grow(std::size_t size)
895 {
896  std::size_t new_capacity = this->capacity_ + this->capacity_ / 2;
897  if (size > new_capacity)
898  new_capacity = size;
899  T *new_ptr = this->allocate(new_capacity, FMT_NULL);
900  // The following code doesn't throw, so the raw pointer above doesn't leak.
901  std::uninitialized_copy(this->ptr_, this->ptr_ + this->size_,
902  make_ptr(new_ptr, new_capacity));
903  std::size_t old_capacity = this->capacity_;
904  T *old_ptr = this->ptr_;
905  this->capacity_ = new_capacity;
906  this->ptr_ = new_ptr;
907  // deallocate may throw (at least in principle), but it doesn't matter since
908  // the buffer already uses the new storage and will deallocate it in case
909  // of exception.
910  if (old_ptr != data_)
911  Allocator::deallocate(old_ptr, old_capacity);
912 }

1074  // and the benchmark https://github.com/localvoid/cxx-benchmark-count-digits.

1139 // Formats a decimal unsigned integer value writing into buffer.
1140 // thousands_sep is a functor that is called after writing each char to
1141 // add a thousands separator if necessary.
1142 template <typename UInt, typename Char, typename ThousandsSep>
1143 inline void format_decimal(Char *buffer, UInt value, unsigned num_digits,
1144  ThousandsSep thousands_sep)
1145 {
1146  buffer += num_digits;
1147  while (value >= 100)
1148  {
1149  // Integer division is slow so do it for a group of two digits instead
1150  // of for every digit. The idea comes from the talk by Alexandrescu
1151  // "Three Optimization Tips for C++". See speed-test for a comparison.
1152  unsigned index = static_cast<unsigned>((value % 100) * 2);
1153  value /= 100;
1154  *--buffer = Data::DIGITS[index + 1];
1155  thousands_sep(buffer);
1156  *--buffer = Data::DIGITS[index];
1157  thousands_sep(buffer);
1158  }
1159  if (value < 10)
1160  {
1161  *--buffer = static_cast<char>('0' + value);
1162  return;
1163  }
1164  unsigned index = static_cast<unsigned>(value * 2);
1165  *--buffer = Data::DIGITS[index + 1];
1166  thousands_sep(buffer);
1167  *--buffer = Data::DIGITS[index];
1168 }
