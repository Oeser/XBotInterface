303 class z_formatter:public flag_formatter
304 {
305 public:
306  const std::chrono::seconds cache_refresh = std::chrono::seconds(5);
307

309  {}

313  void format(details::log_msg& msg, const std::tm& tm_time) override
314  {
315 #ifdef _WIN32
316  int total_minutes = get_cached_offset(msg, tm_time);
317 #else
318  // No need to chache under gcc,
319  // it is very fast (already stored in tm.tm_gmtoff)
320  int total_minutes = os::utc_minutes_offset(tm_time);
321 #endif
322  bool is_negative = total_minutes < 0;
323  char sign;
324  if (is_negative)
325  {
326  total_minutes = -total_minutes;
327  sign = '-';
328  }
329  else
330  {
331  sign = '+';
332  }
333 
334  int h = total_minutes / 60;
335  int m = total_minutes % 60;
336  msg.formatted << sign;
337  pad_n_join(msg.formatted, h, m, ':');
338  }
339 private:
340  log_clock::time_point _last_update;
341  int _offset_minutes;
342  std::mutex _mutex;
343 
344  int get_cached_offset(const log_msg& msg, const std::tm& tm_time)
345  {
346  using namespace std::chrono;
347  std::lock_guard<std::mutex> l(_mutex);
348  if (msg.time - _last_update >= cache_refresh)
349  {
350  _offset_minutes = os::utc_minutes_offset(tm_time);
351  _last_update = msg.time;
352  }
353  return _offset_minutes;
354  }

477 inline void spdlog::pattern_formatter::compile_pattern(const std::string& pattern)
478 {
479  auto end = pattern.end();
480  std::unique_ptr<details::aggregate_formatter> user_chars;
481  for (auto it = pattern.begin(); it != end; ++it)
482  {
483  if (*it == '%')
484  {
485  if (user_chars) //append user chars found so far
486  _formatters.push_back(std::move(user_chars));
487 
488  if (++it != end)
489  handle_flag(*it);
490  else
491  break;
492  }
493  else // chars not following the % sign should be displayed as is
494  {
495  if (!user_chars)
496  user_chars = std::unique_ptr<details::aggregate_formatter>(new details::aggregate_formatter());
497  user_chars->add_ch(*it);
498  }
499  }
500  if (user_chars) //append raw chars found so far
501  {
502  _formatters.push_back(std::move(user_chars));
503  }
